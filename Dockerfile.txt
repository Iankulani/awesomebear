# Multi-stage Dockerfile for Awesome-Bear Cybersecurity Tool
FROM alpine:3.19 as builder

# Install build dependencies
RUN apk add --no-cache \
    python3 \
    py3-pip \
    py3-setuptools \
    py3-wheel \
    gcc \
    musl-dev \
    python3-dev \
    libffi-dev \
    openssl-dev \
    cargo \
    nmap \
    curl \
    bind-tools \
    traceroute \
    openssh-client \
    sudo

# Set working directory
WORKDIR /app

# Copy requirements first for better caching
COPY requirements.txt .
RUN pip3 install --no-cache-dir -r requirements.txt

# Final stage
FROM alpine:3.19

# Install runtime dependencies
RUN apk add --no-cache \
    python3 \
    py3-pip \
    nmap \
    curl \
    bind-tools \
    traceroute \
    openssh-client \
    sudo \
    tcpdump \
    iputils \
    util-linux \
    procps \
    chromium \
    chromium-chromedriver \
    && rm -rf /var/cache/apk/*

# Create non-root user for security
RUN addgroup -g 1000 -S bear && \
    adduser -u 1000 -S bear -G bear && \
    echo "bear ALL=(ALL) NOPASSWD: /sbin/iptables" >> /etc/sudoers

# Copy Python packages from builder
COPY --from=builder /usr/lib/python3*/site-packages /usr/lib/python3*/site-packages
COPY --from=builder /usr/bin /usr/bin

# Create application directories
RUN mkdir -p /app/.awesomebear && \
    mkdir -p /app/reports && \
    chown -R bear:bear /app

# Set working directory
WORKDIR /app

# Copy application files
COPY --chown=bear:bear awesomebear.py .
COPY --chown=bear:bear requirements.txt .

# Create entrypoint script
RUN echo '#!/bin/sh\n\
echo "🐻 Starting Awesome-Bear v1.0.0"\n\
echo "================================="\n\
echo "Container Info:"\n\
echo "  Hostname: $(hostname)"\n\
echo "  IP Address: $(hostname -i 2>/dev/null || echo "unknown")"\n\
echo "================================="\n\
echo "Available Commands:"\n\
echo "  - Execute commands directly: python3 awesomebear.py"\n\
echo "  - Web interface: http://localhost:8080"\n\
echo "=================================\n"\
exec python3 awesomebear.py\n' > /entrypoint.sh && chmod +x /entrypoint.sh

# Expose ports
EXPOSE 8080 8080/tcp
EXPOSE 8080/udp

# Switch to non-root user
USER bear

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
    CMD python3 -c "import socket; s=socket.socket(); s.connect(('localhost', 8080))" || exit 1

# Entry point
ENTRYPOINT ["/entrypoint.sh"]
